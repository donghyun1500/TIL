import sys

input = sys.stdin.readline

a, b = map(int, input().split())


# a!/b!(a-b)!

def factorial(n):
    
    if n > 0:
        
        return n * factorial(n - 1)
    
    else:
        
        return 1


print(factorial(a) // (factorial(b) * factorial(a - b)))
