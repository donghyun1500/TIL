import sys

input = sys.stdin.readline

def factorial(n):
    
    if n > 0:
        
        return n*factorial(n-1)
    
    else:
        
        return 1

n = int(input())

for i in range(n):
    
    m,n = map(int,input().split())
    
    print(factorial(n)//(factorial(m)*factorial(n-m)))
