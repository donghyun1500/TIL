import sys

from collections import deque

class Queue():
    
    def __init__(self):
        
        self.li = deque()

    def push(self,n):
        
        self.li.append(n)

    def pop(self):
        
        print(self.li.popleft() if self.li else -1)

    def size(self):
        
        print(len(self.li))

    def empty(self):
        
        print(0 if self.li else 1)

    def front(self):
        
        print(self.li[0] if self.li else -1)

    def back(self):
        
        print(self.li[-1] if self.li else -1)


queue = Queue()

for i in range(int(sys.stdin.readline())):
    
    a = sys.stdin.readline().split()
    
    if a[0] == "push":
        
        queue.push(int(a[1]))

    elif a[0] == "pop":
        
        queue.pop()

    elif a[0] == "size":
        
        queue.size()

    elif a[0] == "empty":
        
        queue.empty()

    elif a[0] == "front":
        
        queue.front()

    elif a[0] == "back":
        
        queue.back()
