문제풀이 시작!!!!!!!!!!!!!!!!!!!!!


n = int(input())                              #변수n을 통하여 정수를 받고

a = 0                                         #str(a)에 '666'이 들어올때 count를 +1해주고

con = 0                                       #count = n이 같으면 빠져 나오고 

while True:                                   # 아니면 a를 계속해서 +1 더해준다.
    
    if '666' in str(a):
        
        con += 1
    
    if con == n:
        
        break
    
    a += 1


print(a)
